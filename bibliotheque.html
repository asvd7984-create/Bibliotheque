function doPost(e) {
  const s = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  const d = JSON.parse(e.postData.contents);
  if (d.action==='add') {
    s.appendRow([d.isbn,d.title,d.author,d.type,
      d.serie,d.tome,d.publisher,d.year,d.notes,
      new Date().toLocaleDateString('fr-FR')]);
    return ok();
  }
  if (d.action==='delete') {
    const rows=s.getDataRange().getValues();
    for(let i=rows.length-1;i>=1;i--){
      if(rows[i][0]==d.isbn){s.deleteRow(i+1);break;}
    }
    return ok();
  }
  return ok();
}
function doGet(e) {
  const s=SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  const rows=s.getDataRange().getValues().slice(1);
  return ContentService.createTextOutput(
    JSON.stringify({ok:true,rows}))
    .setMimeType(ContentService.MimeType.JSON);
}
function ok(){
  return ContentService.createTextOutput(
    JSON.stringify({ok:true}))
    .setMimeType(ContentService.MimeType.JSON);
}
